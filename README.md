  - Python 3.12
  - NodeJS
    
Backend:
  - cd backend
  - python3 venv venv
  - source venv/bin/activate
  - pip install -r requirements.txt
  - python3 main.py

Frontend:
  - cd Aplicacion_Heladas
  - npm i
  - npm run dev
